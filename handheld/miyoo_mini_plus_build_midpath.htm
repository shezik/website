掌機 - Miyoo Mini Plus - Build MIDPath


參考資訊：
https://github.com/pthalin/bittboy-j2me

步驟如下： 

$ cd
$ wget https://github.com/steward-fu/website/releases/download/miyoo-mini-plus/toolchain.tar.gz
$ tar xvf toolchain.tar.gz
$ sudo mv mmiyoo /opt
$ sudo mv prebuilt /opt
$ export PATH=/opt/mmiyoo/bin/:$PATH

$ cd
$ wget https://github.com/steward-fu/website/releases/download/miyoo-mini-plus/jdk-6u45-linux-i586.bin
$ chmod a+x ./jdk-6u45-linux-i586.bin
$ ./jdk-6u45-linux-i586.bin
$ sudo mkdir -p /usr/lib/jvm/
$ sudo mv jdk1.6.0_45/ /usr/lib/jvm/

$ export PATH=/usr/lib/jvm/jdk1.6.0_45/bin/:$PATH
$ javac -version
    javac 1.6.0_45

$ cd
$ wget https://github.com/steward-fu/website/releases/download/miyoo-mini-plus/src_midpath-rc1.tar.gz
$ tar xvf src_midpath-rc1.tar.gz
$ cd midpath/phoneme_advanced_mr2/cdc/build/linux-arm-miyoo/
$ make
    midpath/phoneme_advanced_mr2/cdc/build/linux-arm-miyoo/./jcs/tbl.cc:148:10: fatal error: tbl.hh: No such file or directory
        148 | #include "tbl.hh"

$ vim ./jcs/tbl.cc +148
    #include "tbl.cc.h"

$ cd 
$ cd midpath/midpath
$ ./build.sh
