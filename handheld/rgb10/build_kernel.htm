1. build_kernel

步驟如下： 

$ sudo apt-get update
$ sudo apt-get install -y git lzop build-essential gcc bc libncurses5-dev libc6-i386 lib32stdc++6 zlib1g:i386 -y

$ cd
$ wget https://releases.linaro.org/components/toolchain/binaries/6.3-2017.05/aarch64-linux-gnu/gcc-linaro-6.3.1-2017.05-x86_64_aarch64-linux-gnu.tar.xz
$ tar xvf gcc-linaro-6.3.1-2017.05-x86_64_aarch64-linux-gnu.tar.xz
$ sudo mv gcc-linaro-6.3.1-2017.05-x86_64_aarch64-linux-gnu /opt/rgb10

$ export ARCH=arm64
$ export CROSS_COMPILE=aarch64-linux-gnu-
$ export PATH=/opt/rgb10/bin/:$PATH

$ git clone https://github.com/hardkernel/linux.git -b odroidgoA-4.4.y
$ cd linux
$ make odroidgo2_defconfig
$ make -j4
