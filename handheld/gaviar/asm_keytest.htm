lcd_wr:
    li a0, GPIO_BASE + PD_DAT

    li t4, 0x00ff
    and t2, t0, t4
    li t4, 0xff00
    and t3, t0, t4
    sll t2, t2, 1
    sll t3, t3, 2

    move t4, t1
    or t4, t4, t2
    or t4, t4, t3
    sw t4, 0(a0)

    li t0, LCD_WR
    or t4, t4, t0
    sw t4, 0(a0)
    jr ra

lcd_dat:
    move a1, ra
    li t1, LCD_RS | LCD_RD | LCD_BL | LCD_RST
    jal lcd_wr
    jr a1

lcd_cmd:
    move a1, ra
    li t1, LCD_RD | LCD_BL | LCD_RST
    jal lcd_wr
    jr a1

    jr a1
