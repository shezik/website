SDL Screen Rotation

main.c 

#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <string.h>
#include <SDL.h>

SDL_Surface *screen = NULL;
SDL_Surface *monitor = NULL;

void rotate_screen(void)
{
    int x = 0, y = 0;
    uint16_t *s = screen->pixels;
    uint16_t *d = monitor->pixels;

    for (y = 0; y < 240; y++) {
        for (x = 0; x < 320; x++) {
            d[((239 - y) * 320) + (319 - x)] = s[(y * 320) + x];
        }
    }
}

int main(int argc, char **argv)
{
    SDL_Rect rt={0};

    SDL_Init(SDL_INIT_VIDEO);
    monitor = SDL_SetVideoMode(320, 240, 16, SDL_SWSURFACE | SDL_DOUBLEBUF);
    screen = SDL_CreateRGBSurface(SDL_SWSURFACE, 320, 240, 16, 0, 0, 0, 0);
    SDL_FillRect(screen, &screen->clip_rect, SDL_MapRGB(screen->format, 0xff, 0x00, 0x00));

    rt.x = 50;
    rt.y = 50;
    rt.w = 30;
    rt.h = 30;
    SDL_FillRect(screen, &rt, SDL_MapRGB(screen->format, 0x00, 0xff, 0x00));

    rt.x = 100;
    rt.y = 100;
    rt.w = 50;
    rt.h = 100;
    SDL_FillRect(screen, &rt, SDL_MapRGB(screen->format, 0x00, 0x00, 0xff));

    rotate_screen();
    SDL_Flip(monitor);
    SDL_Delay(3000);
    SDL_FreeSurface(screen);
    SDL_Quit();
    return 0;
}
