解法如下： 

$ vim arch/arm/Makefile
    arch-$(CONFIG_CPU_32v7M) =-D__LINUX_ARM_ARCH__=7 -march=armv7-m
    arch-$(CONFIG_CPU_32v7)  =-D__LINUX_ARM_ARCH__=7 -march=armv7-a
    arch-$(CONFIG_CPU_32v6)  =-D__LINUX_ARM_ARCH__=6 -march=armv6
