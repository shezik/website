/sbox/login.sh 

#!/bin/sh
if [ `whoami` != "root" ];then
    echo "Run me as root"
    exit 1
fi

MYPATH=`dirname $0`
mount -o bind /proc "$MYPATH/proc"
mount -o bind /sys "$MYPATH/sys"
mount -o bind /dev "$MYPATH/dev"
mount -o bind /dev/pts "$MYPATH/dev/pts"
chroot "$MYPATH" /bin/bash -c 'su - xxx'
umount "$MYPATH/dev/pts"
umount "$MYPATH/dev"
umount "$MYPATH/sys"
umount "$MYPATH/proc"

P.S. xxx是使用者名字
