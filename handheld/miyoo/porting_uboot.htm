+static void sunxi_lcdc_output(uint32_t is_data, uint32_t val)
+{
+  uint32_t ret;
+	struct sunxi_gpio_reg * const gpio = (struct sunxi_gpio_reg *)SUNXI_PIO_BASE;
+
+  ret = (val & 0x00ff) << 1;
+  ret|= (val & 0xff00) << 2;
+  ret|= is_data ? 0x80000 : 0;
+  ret|= 0x100000;
+  writel(ret, &gpio->gpio_bank[SUNXI_GPIO_D].dat);
+  ret|= 0x40000;
+  writel(ret, &gpio->gpio_bank[SUNXI_GPIO_D].dat);
+}
