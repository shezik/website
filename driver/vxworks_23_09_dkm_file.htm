main.c 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <iosLib.h>
 
static int drvNum = 0;
static DEV_HDR hdr = {0};

void* myOpen(struct dev_hdr *dev, const char *remainder, int access, int mode)
{
    if (remainder[0] != '\0') {
        return NULL;
    }

    printf("%s()\n", __func__);
    return dev;
}
 
int myClose(void *dev)
{
    if (dev == NULL) {
        return ERROR;
    }

    printf("%s()\n", __func__);
    return OK;
}

long myRead(void *dev, char *buf, size_t size)
{
    if (dev == NULL) {
        return ERROR;
    }

    printf("%s()\n", __func__);
    return 0;
}

long myWrite(void *dev, const char *buf, size_t size)
{
    if (dev == NULL) {
        return ERROR;
    }

    printf("%s()\n", __func__);
    return 0;
}
 
int myProbe(void)
{
    drvNum = iosDrvInstall(NULL, NULL, myOpen, myClose, myRead, myWrite, NULL);
    iosDevAdd(&hdr, "/dev/hello", drvNum);
    printf("Hello, world!\n");
    return OK;
}
