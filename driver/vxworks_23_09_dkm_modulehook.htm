main.c 

#include <stdio.h>
#include <private/rtpLibP.h>

int create_hook(MODULE_ID moduleId)
{
    printf("%s: %s\n", __func__, moduleNameGet(moduleId));
    return OK;
}

int add_hook(void)
{
    moduleCreateHookAdd(create_hook);
    return OK;
}

編譯 

$ wr-cc main.c -o main -dkm

執行 

-> ld < main

-> add_hook

-> ld < main
    create_hook: main

-> unld "main"
