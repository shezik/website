main.c 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <iosLib.h>
#include <vxWorks.h>
#include <sysLib.h>
#include <wdLib.h>

#define SEC 1

static int val = 0;
static WDOG_ID wid = NULL;

int wdg_handler(void *param)
{
    val += 1;
    wdStart(wid, sysClkRateGet() * SEC, wdg_handler, 0);
    return OK;
}

int start_dkm(void)
{
    wid = wdCreate();
    wdStart(wid, sysClkRateGet() * SEC, wdg_handler, 0);
    return OK;
}

int get_val(void)
{
    printf("%d\n", val);
    return OK;
}

int stop_dkm(void)
{
    wdCancel(wid);
    return OK;
}
