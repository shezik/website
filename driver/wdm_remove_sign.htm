main.cpp 

#include <stdafx.h>
#include <windows.h>
#include <imagehlp.h>

#pragma comment(lib, "Imagehlp.lib")

int main(int argc, char** argv)
{
    HANDLE hFile;

    hFile = CreateFile(argv[1], GENERIC_READ | GENERIC_WRITE, 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
    ImageRemoveCertificate(hFile, 0);
    CloseHandle(hFile);
    return 0;
}
