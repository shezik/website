main.c 

#include <stdio.h>
#include <string.h>
#include <windows.h>
#include <commctrl.h>
 
int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine, int nCmdShow)
{
    char buf[255] = {0};

    LoadString(hInstance, 1001, buf, sizeof(buf));
    MessageBox(NULL, buf, "main", MB_OK);
    ExitProcess(0);
    return 0;
}

編譯、執行

$ winegcc main.c main.res -o main
