參考資訊：
1. msdn
2. start

說明：

API
int MessageBox(HWND hWnd, LPCTSTR lpText, LPCTSTR lpCaption, UINT uType);

main.s 

    .global WinMain
    .extern MessageBoxA
    .extern ExitProcess
 
    .data
title: .asciz "main"
content: .asciz "Hello, world!"
 
    .text
WinMain:
    stp x29, x30, [sp, #-0x10]!
 
    mov x0, 0
    ldr x1, =content
    ldr x2, =title
    mov x3, 0
    bl MessageBoxA

    mov x0, 0
    bl ExitProcess
 
    mov x0, 0
    ldp x29, x30, [sp], #0x10
    ret

編譯、執行 

$ winegcc main.s -o main
$ wine ./main.exe
