main.s 

    .global _start

    .data
msg: .ascii "hello, world!\n"
len = . - msg

    .text
_start:
    mov r0, #1
    ldr r1, =msg
    ldr r2, =len
    swi 0x900004

    mov r0, #0
    swi 0x900001

NRsyscall namearg0(r0)arg1(r1)arg2(r2)
0x900001exitint error_code

編譯、執行 

$ arm-linux-gnueabihf-gcc main.s -o main -nostdlib -static
$ qemu-arm ./main
    hello, world!
