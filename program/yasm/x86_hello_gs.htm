main.s 

    global main
  
    section .data
msg: db "hello, world!", 10
len: equ $ - msg
  
    section .text
main:
    mov eax, 4
    mov ebx, 1
    mov ecx, msg
    mov edx, len
    call [gs:0x10]
    
    mov eax, 1
    mov ebx, 0
    call [gs:0x10]
