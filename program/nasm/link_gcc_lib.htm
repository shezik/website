NASM

如何呼叫gcc library(沒有使用gcc compiler)

參考資訊：
1. nasm

main.s 

  global _start
  extern atol
  extern printf
  extern __uClibc_main

  section .data
num db "100", 0
fmt db "0x%08x", 10, 0

  section .text
_start:
  xor ebp, ebp
  pop esi
  mov ecx, esp
  and esp, 0fffffff0h
  push eax
  push esp
  push edx
  push 0
  push 0
  push ecx
  push esi
  push main
  call __uClibc_main
  xor eax, eax
  leave
  ret

main:
  push ebp
  mov ebp, esp

  push num
  call atol

  push eax
  push fmt
  call printf

  xor eax, eax
  leave
  ret

編譯、執行 

$ nasm -felf32 main.s -o main.o
$ ld -melf_i386 main.o -o main libc.a
$ ./main
  0x00000064
