1. assigning-value-to-variable-in-x86-nasm

main.s 

  global main
  extern printf

  section .data
fmt  db "%d", 0
in_0 dd 0

  section .text
main:
  push ebp
  mov ebp, esp

  mov eax, 1234
  mov [in_0], eax

  push dword [in_0]
  push fmt
  call printf

  xor eax, eax
  leave
  ret

編譯、執行 

$ nasm -felf32 main.s -o main.o
$ gcc main.o -o main -m32
$ ./main
  1234
