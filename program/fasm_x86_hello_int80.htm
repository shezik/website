main.s 

    format elf executable 3
    entry start

    segment readable writeable
msg db "hello, world!", 10
len = $ - msg

    segment readable executable
start:
    mov eax, 4
    mov ebx, 1
    mov ecx, msg
    mov edx, len
    int 0x80
 
    mov eax, 1
    xor ebx, ebx
    int 0x80

編譯、執行 

$ qemu-i386 /usr/local/bin/fasm main.s 
$ qemu-i386 ./main
    hello, world!
