main.c 

%include @Environ(OBJASM_PATH)\Code\Macros\Model.inc
SysSetup OOP, WIN32, ANSI_STRING
   
%includelib &LibPath&Windows\shell32.lib
   
MakeObjects Primer, Stream, WinPrimer, Window, Button, Hyperlink, Dialog, DialogModal, DialogAbout, WinApp, SdiApp

    .const   
Object MyApp, , SdiApp
    RedefineMethod Init
ObjectEnd
   
    .code
Method MyApp.Init, uses esi
    SetObject esi
    invoke CreateWindowEx, WS_EX_LEFT, WC_DIALOG, $OfsCStr("main"), 
        WS_OVERLAPPEDWINDOW or WS_VISIBLE, 0, 0, 300, 300, NULL, NULL, NULL, NULL
    m2m [xsi].hWnd, eax
MethodEnd

start proc
    SysInit
   
    OCall $ObjTmpl(MyApp)::MyApp.Init
    OCall $ObjTmpl(MyApp)::MyApp.Run
    OCall $ObjTmpl(MyApp)::MyApp.Done
   
    SysDone
    invoke ExitProcess, 0
start endp
   
end
