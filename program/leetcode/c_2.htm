struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2)
{
    int c0 = 0, c1 = 0, cc = 0;
    struct ListNode *r = l1 ? l1 : l2 ? l2 : NULL;
    struct ListNode *t = r;

    while (t) {
        if (l1) {
            c0 = l1->val;
            l1 = l1->next;
        }

        if (l2) {
            c1 = l2->val;
            l2 = l2->next;
        }

        t->val = c0 + c1 + cc;
        c0 = 0;
        c1 = 0;
        cc = 0;
        if(t->val >= 10){
            t->val-= 10;
            cc = 1;
        }

        t->next = l1 ? l1 : l2 ? l2 : NULL;
        if (t->next == NULL) {
            break;
        }
        t = t->next;
    }

    if (cc) {
        t->next = malloc(sizeof(struct ListNode));
        t->next->val = cc;
        t->next->next = NULL;
    }
    return r;
}
