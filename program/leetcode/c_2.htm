struct ListNode* addTwoNumbers(struct ListNode *l1, struct ListNode *l2)
{
    struct ListNode *t1=l1;
    struct ListNode *t2=l2;
    struct ListNode *tt1=NULL;
    struct ListNode *tt2=NULL;
    int m1=0, m2=0;
    int c1=0, c2=0, n=0;
    
    while(1){
        c1 = 0;
        c2 = 0;
        if(t1){
            m1+= 1;
            c1 = t1->val;
        }
        
        if(t2){
            m2+= 1;
            c2 = t2->val;
        }
        
        if(t1 || t2){
            if(t1){
                t1->val = c1 + c2 + n;
            }
            if(t2){
                t2->val = c1 + c2 + n;
            }
            n = 0;
        }
        
        if(t1 && (t1->val >= 10)){
            t1->val-= 10;
            n = 1;
        }
        if(t2 && (t2->val >= 10)){
            t2->val-= 10;
            n = 1;
        }
        
        if(t1){
            tt1 = t1;
            t1 = t1->next;
        }
        
        if(t2){
            tt2 = t2;
            t2 = t2->next;
        }

        if((t1 == NULL) && (t2 == NULL)){
            if(n > 0){
                if(m1 > m2){
                    tt1->next = l2;
                    tt1->next->val = n;
                    tt1->next->next = NULL;
                }
                else{
                    tt2->next = l1;
                    tt2->next->val = n;
                    tt2->next->next = NULL;
                }
            }
            break;
        }
    }
    
    if(m1 > m2){
        return l1;
    }
    return l2;
}
