impl Solution {
    pub fn sort_array(nums: Vec<i32>) -> Vec<i32> {
        let l = nums.len();
        let mut v = vec![0; 100001];
        let mut r = Vec::new();

        for i in 0..l {
            v[nums[i] as usize + 50000]+= 1;
        }
        for i in 0..100001 {
            while v[i] > 0 {
                r.push(i as i32 - 50000);
                v[i]-= 1;
            }
        }
        return r;
    }
}
