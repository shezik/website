typedef struct{
    int pos;
    int num;
}pt;

int comp(const void *s0, const void *s1)
{
    pt *p0=(pt*)s0;
    pt *p1=(pt*)s1;
    
    if(p0->num > p1->num){
        return 1;
    }
    if(p0->num < p1->num){
        return -1;
    }
    return 0;
}

bool containsNearbyDuplicate(int *nums, int numsSize, int k)
{
    int c0=0;
    pt *p=malloc(sizeof(pt) * numsSize);
    
    for(c0=0; c0<numsSize; c0++){
        p[c0].num = nums[c0];
        p[c0].pos = c0;
    }
    
    qsort(p, numsSize, sizeof(pt), comp);
    for(c0=0; c0<numsSize-1; c0++){
        if(p[c0].num == p[c0+1].num){
            if(abs(p[c0].pos - p[c0+1].pos) <= k){
                free(p);
                return true;
            }
        }
    }
    free(p);
    return false;
}
