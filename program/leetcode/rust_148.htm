impl Solution {
    pub fn sort_list(head: Option<Box<ListNode>>) -> Option<Box<ListNode>> {
        if head.as_ref().is_none() {
            return head;
        }

        let mut v = Vec::new();
        let mut t0 = head.as_ref();
        while let Some(node) = t0 {
            v.push(node.val);
            t0 = node.next.as_ref();
        }
        v.sort();

        let mut r0 = Some(Box::new(ListNode{val: v[0], next: None}));
        let mut r1 = &mut r0;
        for i in 1..v.len() {
            let mut t = r1.as_mut().unwrap();
            t.next = Some(Box::new(ListNode{val: v[i], next: None}));
            r1 = &mut t.next;
        }
        return r0;
    }
}
