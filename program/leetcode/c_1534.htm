1534. Count Good Triplets

int countGoodTriplets(int *arr, int arrSize, int a, int b, int c)
{
    int c0=0, c1=0, c2=0, r=0, tmp=0;
    
    for(c0=0; c0<arrSize-2; c0++){
        for(c1=c0+1; c1<arrSize-1; c1++){
            tmp = abs(arr[c0] - arr[c1]);
            if(tmp > a){
                continue;
            }
            for(c2=c1+1; c2<arrSize; c2++){
                tmp = abs(arr[c1] - arr[c2]);
                if(tmp > b){
                    continue;
                }
                
                tmp = abs(arr[c2] - arr[c0]);
                if(tmp > c){
                    continue;
                }
                r+= 1;
            }
        }
    }
    return r;
}
