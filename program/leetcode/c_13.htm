int toInt(char ch)
{
    if(ch == 'I'){
        return 1;
    }
    else if(ch == 'V'){
        return 5;
    }
    else if(ch == 'X'){
        return 10;
    }
    else if(ch == 'L'){
        return 50;
    }
    else if(ch == 'C'){
        return 100;
    }
    else if(ch == 'D'){
        return 500;
    }
    else if(ch == 'M'){
        return 1000;
    }
    return -1;
}

int romanToInt(char *s)
{
    int len=strlen(s);
    int c0=0, t0=0, t1=0, r=0;
    
    for(c0=0; c0<len; c0++){
        t0 = toInt(s[c0]);
        if(c0 != (len - 1)){
            t1 = toInt(s[c0 + 1]);
            if(t0 < t1){
                r+= (t1 - t0);
                c0+= 1;
            }
            else{
                r+= t0;
            }
        }
        else{
            r+= t0;
        }
    }
    return r;
}
