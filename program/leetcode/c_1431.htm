1431. Kids With the Greatest Number of Candies

bool* kidsWithCandies(int *candies, int candiesSize, int extraCandies, int *returnSize)
{
    int c0=0, c1=0, ex=0, cnt=0;
    bool *p=malloc(sizeof(bool) * candiesSize);
    
    *returnSize = 0;
    for(c0=0; c0<candiesSize; c0++){
        cnt = 1;
        ex = candies[c0] + extraCandies;
        for(c1=0; c1<candiesSize; c1++){
            if(ex < candies[c1]){
                cnt = 0;
                break;
            }
        }
        if(cnt > 0){
            p[(*returnSize)++] = true;
        }
        else{
            p[(*returnSize)++] = false;
        }
    }
    return p;
}
