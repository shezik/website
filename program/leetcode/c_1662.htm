1662. Check If Two String Arrays are Equivalent

答案： 

bool arrayStringsAreEqual(char **word1, int word1Size, char **word2, int word2Size)
{
    int w1[2]={0}, w2[2]={0};
    
    while((w1[0] < word1Size) && (w2[0] < word2Size)){
        if(word1[w1[0]][w1[1]] != word2[w2[0]][w2[1]]){
            return false;
        }
        
        w1[1]+= 1;
        if(w1[1] >= strlen(word1[w1[0]])){
            w1[0]+= 1;
            w1[1] = 0;
        }
        w2[1]+= 1;
        if(w2[1] >= strlen(word2[w2[0]])){
            w2[0]+= 1;
            w2[1] = 0;
        }
    }
    
    return ((w1[0] >= word1Size) && (w2[0] >= word2Size));
}
