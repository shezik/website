int* intersection(int *nums1, int nums1Size, int *nums2, int nums2Size, int *returnSize)
{
    int c0=0;
    int *tmp=malloc(sizeof(int) * 1001);
    memset(tmp, 0, sizeof(int) * 1001);
    
    for(c0=0; c0<nums1Size; c0++){
        tmp[nums1[c0]]|= 1;
    }
    for(c0=0; c0<nums2Size; c0++){
        tmp[nums2[c0]]|= 2;
    }
    
    *returnSize = 0;
    int *p=malloc(sizeof(int) * 1001);
    for(c0=0; c0<1001; c0++){
        if(tmp[c0] == 3){
            p[(*returnSize)++] = c0;
        }
    }
    free(tmp);
    return p;
}
