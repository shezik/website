int oddCells(int m, int n, int **indices, int indicesSize, int *indicesColSize)
{
    int c0=0, c1=0, c2=0;
    int **p=malloc(sizeof(int*) * m);
    
    for(c0=0; c0<m; c0++){
        p[c0] = malloc(sizeof(int) * n);
        memset(p[c0], 0, sizeof(int) * n);
    }
    
    for(c0=0; c0<indicesSize; c0++){
        for(c2=0; c2<n; c2++){
            p[indices[c0][0]][c2]+= 1;
        }

        for(c2=0; c2<m; c2++){
            p[c2][indices[c0][1]]+= 1;
        }
    }
    
    int r=0;
    for(c0=0; c0<m; c0++){
        for(c1=0; c1<n; c1++){
            if(p[c0][c1] % 2){
                r+= 1;
            }
        }
        free(p[c0]);
    }
    free(p);
    return r;
}
