5. Longest Palindromic Substring

void searchPalindrome(const char *s, int l, int r, int *start, int *cnt)
{
    int len = strlen(s);

    while ((l >= 0) && (l < len) && (s[l] == s[r])) {
        l-= 1;
        r+= 1;
    }
    if (*cnt < (r - l - 1)) {
        *start = l + 1;
        *cnt = r - l - 1;
    }
}

char* longestPalindrome(char *s)
{
    int len = strlen(s);
    int c0 = 0, cnt = 0, start = 0;

    if (len < 2) {
        return s;
    }

    for (c0 = 0; c0 < len - 1; c0++) {
        searchPalindrome(s, c0, c0, &start, &cnt);
        searchPalindrome(s, c0, c0 + 1, &start, &cnt);
    }

    s[start + cnt] = 0;
    return &s[start];
}
