main.s 

    format elf executable 3
    entry start

    segment readable executable
start:
    mov r0, 1
    adr r1, msg
    mov r2, len
    swi 0x900004

    mov r0, 0
    swi 0x900001

msg: db "hello, world!", 10
len = $ - msg

0x900001exitint error_code

編譯、執行 

$ qemu-i386 /usr/local/bin/fasmarm main.s 
$ qemu-arm ./main
    hello, world!
