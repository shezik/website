main.asm 

    %include "head.asm"
 
    segment .text
WndProc:
    push ebp
    mov ebp, esp
   
    cmp dword [ebp + ARG2], WM_MOUSEMOVE
    je .handle_mouse
    cmp dword [ebp + ARG2], WM_CLOSE
    je .handle_close
    cmp dword [ebp + ARG2], WM_DESTROY
    je .handle_destroy
    jmp .handle_default
  
.handle_mouse:
    mov eax, [ebp + ARG4]
    and eax, 0ffffh
    push eax
    mov eax, [ebp + ARG4]
    shr eax, 16
    and eax, 0ffffh
    push eax
    push FMT_MOUSE
    push pBuf
    call sprintf
  
    push pBuf
    push dword [ebp + ARG1]
    call SetWindowText
    xor eax, eax
    jmp .finish
  
.handle_close:
    push dword [ebp + ARG1]
    call DestroyWindow
    xor eax, eax
    jmp .finish
   
.handle_destroy:
    push 0
    call PostQuitMessage
    xor eax, eax
    jmp .finish
   
.handle_default:
    push dword [ebp + ARG4]
    push dword [ebp + ARG3]
    push dword [ebp + ARG2]
    push dword [ebp + ARG1]
    push dword [pDefWndProc]
    call CallWindowProc
   
.finish:
    leave
    ret 16
   
WinMain:
    push ebp
    mov ebp, esp
    
    push 0
    push 0
    push 0
    push 0
    push 300
    push 300
    push 0
    push 0
    push WS_OVERLAPPEDWINDOW | WS_VISIBLE
    push szAppName
    push WC_DIALOG
    push WS_EX_LEFT
    call CreateWindowEx
    mov [hWin], eax

    push WndProc
    push GWL_WNDPROC
    push dword [hWin]
    call SetWindowLong
    mov [pDefWndProc], eax

.loop:
    push 0
    push 0
    push 0
    push msg
    call GetMessage
    cmp eax, 0
    je .exit
    
    push msg
    call DispatchMessage
    jmp .loop
    
.exit:
    mov eax, [msg + MSG.wParam]
    leave
    ret 16
    
_start:
    push 0
    call GetModuleHandle
    mov [hInstance], eax
    
    call GetCommandLine
    mov [pCommand], eax
    
    push SW_SHOWNORMAL
    push dword [pCommand]
    push 0
    push dword [hInstance]
    call WinMain
    
    push eax
    call ExitProcess
