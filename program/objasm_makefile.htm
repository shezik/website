export WINEPREFIX=/home/user/.wine_amd64

TARGET=main
MYWINE=box86 wine
OBJASM=/home/user/.wine_amd64/drive_c/ObjAsm

all:
	${MYWINE} ${OBJASM}/Build/Tools/Assembler/uasm64.exe @c:/ObjAsm/Build/Options/OPT_ASM_RLS_64.txt ${TARGET}.asm
	${MYWINE} ${OBJASM}/Build/Tools/Linker/link.exe @c:/ObjAsm/Build/Options/OPT_LNK_RLS_64.txt ${TARGET}.obj

run:
	${MYWINE} ${TARGET}.exe

clean:
	rm -rf ${TARGET}.exe ${TARGET}.obj
