use std::cmp;

impl Solution {
    pub fn length_of_longest_substring(s: String) -> i32 {
        let mut r = 0;
        let mut left = -1;
        let mut hash: Vec<i32> = vec![-1; 128];

        for (pos, ch) in s.char_indices() {
            left = cmp::max(left, hash[ch as usize]);
            hash[ch as usize] = pos as i32;
            r = cmp::max(r, (pos as i32 - left));
        }
        return r;
    }
}
