$ cd
$ wget https://github.com/steward-fu/website/releases/download/bcb6/bcb6_wine_patch.tar.gz
$ tar xvf bcb6_wine_patch.tar.gz
$ cp bcb6bins_patched/* /home/user/.wine_amd64/drive_c/Program\ Files\ \(x86\)/Borland/CBuilder6/Bin/
