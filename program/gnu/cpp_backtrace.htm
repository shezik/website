main.c 

#include <execinfo.h>
#include <fcntl.h>
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

void bt(void)
{
    int s = 0;
    int fd = -1;
    void *b[255] = {0};

    fd = open("bt.txt", O_CREAT | O_WRONLY, 0755);
    if (fd > 0) {
        s = backtrace(b, 255);
        backtrace_symbols_fd(b, s, fd);
        close(fd);
    }
}

void test(void)
{
    bt();
}

int main(int argc, char **argv)
{
    test();
}
