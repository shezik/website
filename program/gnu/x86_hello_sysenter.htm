1. how-to-use-sysenter-under-linux

main.s 

  .global _start
  
  .text
_start:
  push %ebp
  mov %esp, %ebp
  mov $msg_len, %edx
  mov $msg, %ecx
  mov $1, %ebx
  mov $4, %eax
  push $ret
  push %ecx
  push %edx
  push %ebp
  mov %esp, %ebp
  sysenter

ret:
  mov $0, %ebx
  mov $1, %eax
  push $ret
  push %ecx
  push %edx
  push %ebp
  mov %esp, %ebp
  sysenter
  
  .data
msg: .ascii "hello, world!\n"
msg_len = . - msg

編譯、執行 

$ as --32 main.s -o main.o
$ ld -m elf_i386 main.o -o main
$ ./main
  hello, world!
