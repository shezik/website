程式語言 - Simple DirectMedia Layer (SDL) - v1.2 - C/C++ - Load PNG from Memory


參考資訊：
https://www.libsdl.org/release/SDL-1.2.15/docs/html/index.html

main.c 

#include <stdio.h>
#include <stdlib.h>
#include <SDL.h>
#include <SDL_image.h>

#include "hex_png.h"

int main(int argc, char **argv)
{
    SDL_Surface *screen = NULL;

    SDL_Init(SDL_INIT_VIDEO);
    screen = SDL_SetVideoMode(320, 240, 16, SDL_HWSURFACE);

    SDL_RWops *rw = SDL_RWFromMem(hex_png, sizeof(hex_png));
    SDL_Surface *png = IMG_Load_RW(rw, 1);
    SDL_BlitSurface(png, NULL, screen, NULL);
    SDL_Flip(screen);
    SDL_Delay(3000);

    SDL_FreeSurface(png);
    SDL_Quit();
    return 0;
}

編譯、執行 

$ gcc main.c -o main -lSDL -lSDL_image -I/usr/include/SDL
$ ./main
