解答： 

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int r = 0;
        int left = -1;
        vector<int> hash(128, -1);

        for (int c0 = 0; c0 < s.size(); c0++) {
            left = max(left, hash[s[c0]]);
            hash[s[c0]] = c0;
            r = max(r, (c0 - left));
        }
        return r;
    }
};
