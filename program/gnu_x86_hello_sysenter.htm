main.s 

    .global _start
    
    .data
msg: .ascii "hello, world!\n"
len = . - msg
  
    .text
_start:
    push %ebp
    mov %esp, %ebp

    mov $4, %eax
    mov $1, %ebx
    mov $msg, %ecx
    mov $len, %edx
    push $final
    push %ecx
    push %edx
    push %ebp
    mov %esp, %ebp
    sysenter

final:
    mov $1, %eax
    mov $0, %ebx
    push $final
    push %ecx
    push %edx
    push %ebp
    mov %esp, %ebp
    sysenter
