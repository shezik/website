main.c 

#include <stdio.h>
#include <wayland-client.h>

int main(int argc, char **argv)
{
    struct wl_display *display = wl_display_connect(NULL);

    if (display == NULL) {
        printf("failed to connect !\n");
        return -1;
    }
    printf("connected !\n");
    wl_display_disconnect(display);
    return 0;
}

編譯、執行 

$ gcc main.c -o main -lwayland-client
$ ./main
    connected !
