main.c 

#include <stdio.h>
#include <string.h>
#include <wayland-client.h>

struct wl_compositor *comp = NULL;

void cb_handle(void *dat, struct wl_registry *reg, uint32_t name, const char *intf, uint32_t ver)
{
    if (strcmp(intf, "wl_compositor") == 0) {
        comp = wl_registry_bind(reg, name, &wl_compositor_interface, 3);
    }
}

void cb_remove(void *dat, struct wl_registry *reg, uint32_t name)
{
}

int main(int argc, char **argv)
{
    struct wl_display *dis = wl_display_connect(NULL);
    struct wl_registry *reg = wl_display_get_registry(dis);
    struct wl_registry_listener cb = {
        .global = cb_handle,
        .global_remove = cb_remove
    };

    wl_registry_add_listener(reg, &cb, NULL);
    wl_display_roundtrip(dis); // wait for ready

    if (comp != NULL) {
        printf("bind wl_compositor successfully !\n");
    }
    else {
        printf("failed to bind wl_compositor !\n");
    }

    while (1) {
        wl_display_dispatch(dis);
    }
}
