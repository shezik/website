main.s 

    .global main
 
    .data
msg: .ascii "hello, world!\n"
len = . - msg
 
    .text
main:
    mov $4, %eax
    mov $1, %ebx
    mov $msg, %ecx
    mov $len, %edx
    call *%gs:0x10
   
    mov $1, %eax
    mov $0, %ebx
    call *%gs:0x10
