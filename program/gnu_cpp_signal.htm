main.c 

#include <stdio.h>
#include <unistd.h>
#include <signal.h>
 
void handle_signal(int sig)
{
    printf("User break\n");
    signal(SIGINT, SIG_DFL);
}
 
int main(int argc, char *argv[])
{
    signal(SIGINT, handle_signal);
    pause();
    return 0;
}
