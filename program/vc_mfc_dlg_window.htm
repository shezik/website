程式語言 - Visual C++ 6.0 - Microsoft Fundation Class (MFC) - Dialog - Create Window


main.cpp 

#include <afxwin.h>
#include "res.h"
 
class CMyDlg : public CDialog
{
public:
    CMyDlg();
    enum { IDD = IDD_MAIN_DIALOG };
};
    
CMyDlg::CMyDlg() : CDialog(CMyDlg::IDD)
{
}
 
class CMyApp : public CWinApp
{
public:
    BOOL InitInstance();
};
 
CMyApp theApp;

BOOL CMyApp::InitInstance()
{
    CMyDlg dlg;

    m_pMainWnd = &dlg;
    dlg.DoModal();
    return FALSE;
}

main.rc 

#include "res.h"
#include "afxres.h"

IDD_MAIN_DIALOG DIALOGEX  0, 0, 150, 100 
STYLE DS_MODALFRAME | WS_VISIBLE
EXSTYLE WS_EX_APPWINDOW

BEGIN
END
