程式語言 - Simple DirectMedia Layer (SDL) - v1.2 - C/C++ - Load Music


參考資訊：
https://www.libsdl.org/release/SDL-1.2.15/docs/html/index.html

main.c 

#include <stdio.h>
#include <stdlib.h>
#include <SDL.h>
#include <SDL_mixer.h>

int main(int argc, char **argv)
{
    SDL_Init(SDL_INIT_AUDIO);
    Mix_OpenAudio(44100, MIX_DEFAULT_FORMAT, 2, 4096);

    Mix_Music *music = Mix_LoadMUS("main.wav");
    Mix_PlayMusic(music, 1);
    SDL_Delay(3000);

    Mix_HaltMusic();
    Mix_FreeMusic(music);
    Mix_CloseAudio();
    SDL_Quit();
    return 0;
}

P.S. 支援WAVE, MOD, MIDI, OGG, MP3, FLAC

編譯 

$ gcc main.c -o main -lSDL -lSDL_mixer -I/usr/include/SDL
