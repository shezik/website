module(0): palette
  instance(0): brightness
  instance(0): colors

  instance(0): R2
  instance(0): SP
  instance(0): int_state
  instance(0): irq
  instance(0): irq_max
  instance(0): irq_state
  instance(0): nmi_state
  instance(0): request_irq
  instance(0): service_irq
  instance(1): AF
