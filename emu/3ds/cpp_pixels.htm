main.c 

#include <3ds.h>
#include <stdio.h>
#include <string.h>

int main(void)
{
    u8 *fb = NULL;
    int x = 0, y = 0;

    gfxInitDefault();
    gfxSetDoubleBuffering(GFX_BOTTOM, false);

    fb = gfxGetFramebuffer(GFX_BOTTOM, GFX_LEFT, NULL, NULL);
    for (y = 0; y < 240; y++) {
        for (x = 0; x < 320; x++) {
            fb[0] = 0x00;
            fb[1] = 0x00;
            fb[2] = 0xff;
            fb+= 3;
        }
    }

    gfxFlushBuffers();
    gfxSwapBuffers();
    gspWaitForVBlank();
    svcSleepThread(3000000000LL);
    gfxExit();
    return 0;
}
