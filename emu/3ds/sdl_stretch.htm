Stretch Surface

參考資訊：
1. libctru doc
2. 3ds-examples

說明：

API
int SDL_SoftStretch(SDL_Surface *src, const SDL_Rect *srcrect, SDL_Surface *dst, const SDL_Rect *dstrect)

main.c 

#include <stdio.h>
#include <stdlib.h>
#include <SDL.h>
 
int main(void)
{
    SDL_Surface *screen = NULL;

    SDL_Init(SDL_INIT_VIDEO);
    screen = SDL_SetVideoMode(400, 240, 16, SDL_HWSURFACE);

    SDL_Surface* bmp = SDL_LoadBMP("sdmc:/3ds/main.bmp");
    SDL_SoftStretch(bmp, NULL, screen, NULL);
    SDL_Flip(screen);
    SDL_Delay(3000);

    SDL_FreeSurface(bmp);
    SDL_Quit();
    return 0;
}

P.S. 圖片像素格式必須跟SDL_SetVideoMode()一致，可以使用ffmpeg -i src.jpg -pix_fmt rgb565 -y dst.bmp轉換或者使用SDL_ConvertSurface()轉換
