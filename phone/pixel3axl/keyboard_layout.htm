司徒目前使用queekboard-terminal-layout作為Terminal的Keyboard Layout，然後，新增一個Custom Layout作為觸控遊戲手把使用(~/.local/share/squeekboard/keyboards/custom.yaml) 

---
outlines:
    default: { width: 100, height: 100 }
    action: { width: 200, height: 100 }

views:
    base:
        - "L d1 R d2 F d1 preferences"
        - "d1 ↑ d1 0 1 d1 X d1"
        - "← d1 → d2 Y d1 A"
        - "d1 ↓ d1 N M d1 B d1"

buttons:
    d1:
        action:
            set_view: "base"
        outline: "default"
        label: ""
    d2:
        action:
            set_view: "base"
        outline: "action"
        label: ""
    preferences:
        action: "show_prefs"
        outline: "default"
        icon: "keyboard-mode-symbolic"
    "↑":
        outline: "default"
        keysym: "Up"
    "↓":
        outline: "default"
        keysym: "Down"
    "←":
        outline: "default"
        keysym: "Left"
    "→":
        outline: "default"
        keysym: "Right"



~/.local/share/squeekboard/keyboards/terminal/us.yaml 

---
outlines:
    default: { width: 48, height: 64 }
    menu: { width: 98, height: 64 }
    altline: { width: 68, height: 64 }
    spaceline: { width: 165, height: 64 }
    special: { width: 38, height: 64 }
    action: { width: 48, height: 64 }
    fkey: { width: 40, height: 64 }

views:
    base:
        - "1 2 3 4 5 6 7 8 9 0 BackSpace"
        - "q w e r t y u i o p show_symbols"
        - "a s d f g h j k l Return"
        - "Shift_L   z x c v b n m , ↑ ."
        - "Ctrl Tab show_eschars preferences         space ← ↓ →"
    upper:
        - "! @ # $ % ^ & * ( ) BackSpace"
        - "Q W E R T Y U I O P show_symbols"
        - "A S D F G H J K L Return"
        - "Shift_L   Z X C V B N M  - PgUp /"
        - "Alt Esc show_eschars preferences         space  Home PgDn End"
    symbols:
        - "~ ` ´ - _ + = ° © ® BackSpace"
        - "[ ] { } \\ | ÷ × √ show_letters"
        - "; : ' \" · € £ ¥ Return"
        - "Del / ? § π τ µ ¶ < ↑ >"
        - "Ctrl Tab show_eschars preferences   space  ← ↓ →"
    eschars:
        - "F1 F2 F3 F4 F5 F6 F7 F8 F9 F10 F11 F12 BackSpace"
        - "ä è é ö ü Ä È É Ö Ü show_symbols"
        - "à â ê î ô À Â Ê Î Ô Return"
        - "Del  « » ç Ç æ œ ß  „ PgUp “"
        - "Alt Esc show_letters preferences         space   Home PgDn End"

                lock_view: "upper"
