import os
import sys
import pyhidra

with pyhidra.open_program(sys.argv[1]) as flat_api:
    program = flat_api.getCurrentProgram()
    listing = program.getListing()
    print('Program: {}'.format(flat_api.currentProgram))
    print('Entry: {}'.format(flat_api.firstFunction))
