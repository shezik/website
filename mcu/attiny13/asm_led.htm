ATtiny13 >> Assembly

LED

DDRB(I/O方向)、PORTB(I/O輸出)


main.s 

  .equ PINB,  0x16
  .equ DDRB,  0x17
  .equ PORTB, 0x18
 
  .org 0x0000
  rjmp main
 
  .org 0x0010
main:
  sbi DDRB, 1
 
loop:
  sbi PORTB, 1
  rcall delay
  cbi PORTB, 1
  rcall delay
  rjmp loop
 
delay:
  ldi r24, 0xff
  ldi r25, 0x80
d0:
  sbiw r24, 1
  brne d0
  ret
