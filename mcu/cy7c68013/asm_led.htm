main.s 

  oeb set 0xb3
  oed set 0xb5

  .org 0h
  jmp _start

  .org 0x100
_start:
  mov oeb, #0x04
  mov oed, #0x00
loop:
  setb p1.2
  call delay
  clr p1.2
  call delay
  jmp loop

delay:
  mov r6, #10
d0:
  mov r5, #255
d1:
  mov r4, #255
d2:
  djnz r4, d2
  djnz r5, d1
  djnz r6, d0
  ret
  .end
