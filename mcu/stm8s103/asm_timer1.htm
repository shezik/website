  .equ TIM1_SR1,   0x5255

  .equ TIM1_CNTRL, 0x525f

main:
  mov PB_DDR, #0x20
  mov PB_CR1, #0x20 
  mov TIM1_PSCRH, #0x00
  mov TIM1_PSCRL, #0x00
  mov TIM1_ARRH, #0xff
  mov TIM1_ARRL, #0xff
  mov TIM1_CNTRH, #0x00
  mov TIM1_CNTRL, #0x00
  mov TIM1_IER, #0x01
  mov TIM1_SR1, #0x01
  mov TIM1_CR1, #0x91
  rim
loop:
  jp loop

timer1_handler:
  mov TIM1_SR1, #0
  bcpl PB_ODR, #5
  iret
