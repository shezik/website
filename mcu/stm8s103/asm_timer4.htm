main:
  mov PB_DDR, #0x20
  mov PB_CR1, #0x20 
  mov TIM4_PSCR, #0x00
  mov TIM4_ARR, #0xff
  mov TIM4_CNTR, #0x00
  mov TIM4_IER, #0x01
  mov TIM4_SR, #0x01
  mov TIM4_CR1, #0x81
  bset PB_ODR, #5
  mov cnt, #0x0000
  rim
loop:
  jp loop

timer4_handler:
  mov TIM4_SR, #0
  ldw x, cnt
  incw x
  ldw cnt, x
  cpw x, #1000
  jrne exit
  mov cnt, #0x0000
  bcpl PB_ODR, #5
exit:
  iret
