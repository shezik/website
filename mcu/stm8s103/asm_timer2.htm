  .equ PB_ODR,     0x5005
  .equ PB_IDR,     0x5006

  .equ TIM2_ARRH,  0x530f
  .equ TIM2_ARRL,  0x5310

main:
  mov PB_DDR, #0x20
  mov PB_CR1, #0x20 
  mov TIM2_PSCR, #0x00
  mov TIM2_ARRH, #0xff
  mov TIM2_ARRL, #0xff
  mov TIM2_CNTRH, #0x00
  mov TIM2_CNTRL, #0x00
  mov TIM2_IER, #0x01
  mov TIM2_SR1, #0x01
  mov TIM2_CR1, #0x81
  rim
loop:
  jp loop

timer2_handler:
  mov TIM2_SR1, #0
  bcpl PB_ODR, #5
  iret
