$ git clone https://github.com/xboot/xboot
$ cd xboot
$ make CROSS_COMPILE=/opt/f133/bin/riscv64-unknown-linux-gnu- PLATFORM=riscv64-f133
$ xfel ddr f133
$ xfel write 0x40000000 output/xboot.bin
$ xfel exec 0x40000000
