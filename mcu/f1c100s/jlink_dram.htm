w4 0x01c20060,0x01004000
w4 0x01c202c0,0x01000000

mem32 0x80000000, 1

C:\> JLink.exe -device arm9 -if JTAG -speed 4000 -jtagconf -1,-1 -autoconnect 1 -CommanderScript C:\main.jlink
