DDRA


PTA


main.asm 

    PTA  .equ 0x00
    DDRA .equ 0x04

    .module main
    .optsdcc -mhc08

    .area codeivt (abs)
    .org 0xfffe
    .dw main

    .area cseg (abs, code)
    .org 0xdc00

main:
    lda #0x02
    sta DDRA

loop:
    eor #0x02
    sta PTA

    ldhx #0xffff
d0:
    aix #-1
    cphx #0x0000
    bne d0

    jmp loop

Makefile 

all:
	sdas6808 -los main.asm
	sdld -s main

flash:
	bl08 -c /dev/ttyUSB0 -b 9600 -t jb8 -x 12 -f -g reset main.s19

clean:
	rm -rf main.lst  main.rel  main.s19  main.sym
