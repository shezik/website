main.asm 

    PTA  .equ 0x00
    DDRA .equ 0x04
    PTE  .equ 0x08
    DDRE .equ 0x09
    POCR .equ 0x1d

    .module main
    .optsdcc -mhc08

    .area codeivt (abs)
    .org 0xfffe
    .dw main

    .area cseg (abs, code)
    .org 0xdc00
main:
    lda #0x10
    sta POCR

    lda #0x02
    sta DDRA

    lda #0x00
    sta DDRE
loop:
    lda PTE
    lsra
    lsra
    lsra
    sta PTA
    jmp loop
