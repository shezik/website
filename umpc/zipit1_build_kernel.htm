微電腦 - Zipit Z1 - Build Kernel 2.4.21


$ cd
$ wget https://github.com/steward-fu/website/releases/download/zipit-z1/cross-3.2.tar.gz
$ 7za x cross-3.2.tar.gz
$ sudo mv usr/local/arm /opt/zipit1
$ export PATH=$PATH:/opt/zipit1/bin

$ wget https://github.com/steward-fu/website/releases/download/zipit-z1/kernel.7z
$ 7za x kernel.7z
$ cd kernel
$ tar xvf linux-2.4.21.tar.gz
$ cd linux-2.4.21
$ zcat ../patch-2.4.21-rmk1.gz | patch -p1
$ patch -p1 < ../patch-2.4.21-rmk1-zipit
$ cp arch/arm/defconfig .config
$ vim Makefile +22
    CROSS_COMPILE = arm-linux-

$ ARCH=arm make menuconfig
$ ARCH=arm make clean
$ ARCH=arm make zImage
